4.3 遥控器按键处理（VodPlayerScreen.kt）
kotlin
Modifier.onKeyEvent { keyEvent ->
    if (keyEvent.action == KeyEvent.ACTION_DOWN) {
        when (keyEvent.key) {
            Key.DpadLeft -> viewModel.switchEpisode(-1)
            Key.DpadRight -> viewModel.switchEpisode(1)
            Key.DpadUp -> viewModel.switchRecommend(-1)
            Key.DpadDown -> viewModel.switchRecommend(1)
            else -> return@onKeyEvent false
        }
        true
    } else false
}
4.4 自动播放启动逻辑（MainActivity.kt）
kotlin
LaunchedEffect(Unit) {
    val settings = SettingsManager(context)
    if (settings.autoResumeLast) {
        val historyDao = AppDatabase.getInstance(context).historyDao()
        val last = historyDao.getAll().firstOrNull()?.firstOrNull()
        when (last?.type) {
            "live" -> currentScreen = Screen.LiveRoom(last.roomId)
            "vod" -> currentScreen = Screen.VodPlayer(last.bvid)
        }
    }
}

点播推荐列表可调用 B站官方接口 x/web-interface/related?bvid=xxx
